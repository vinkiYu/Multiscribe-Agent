### T1：P1-5 慢查询监控（`infra/db.py`）

**问题定位**：`Database.execute/fetchall/fetchone` 无计时，慢查询无感知。

**修复**：

```python
# src/multiscribe_agent/infra/db.py  在 Database 类的查询方法外层包装：

import time
import structlog

log = structlog.get_logger(__name__)


class Database:
    def __init__(self, ..., slow_query_threshold: float = 1.0):
        ...
        self._slow_threshold = slow_query_threshold

    async def execute(self, statement: str, parameters=()):
        started = time.monotonic()
        try:
            result = await self._conn.execute(statement, parameters)
            return result
        finally:
            self._maybe_log_slow(statement, parameters, time.monotonic() - started)

    async def fetchall(self, statement, parameters=()):
        started = time.monotonic()
        try:
            return await self._conn.fetchall(statement, parameters)
        finally:
            self._maybe_log_slow(statement, parameters, time.monotonic() - started)

    # fetchone 同理...

    def _maybe_log_slow(self, statement: str, parameters, duration: float) -> None:
        if duration < self._slow_threshold:
            return
        log.warning(
            "slow_query",
            statement=statement[:200],
            param_count=len(parameters) if hasattr(parameters, "__len__") else 0,
            duration_ms=round(duration * 1000, 2),
            threshold_ms=round(self._slow_threshold * 1000, 2),
        )
        try:
            from multiscribe_agent.observability.meter import get_metrics_registry
            get_metrics_registry().record_slow_query(duration)
        except Exception:
            pass  # observability 不可用时不影响主流程
```

```python
# src/multiscribe_agent/observability/sql_audit.py
"""Append-only SQL audit log with suspicious-pattern detection."""

from __future__ import annotations

import re
import time
from dataclasses import dataclass

from multiscribe_agent.infra.db import Database


SUSPICIOUS_PATTERNS = [
    re.compile(r"\bDROP\b\s", re.IGNORECASE),
    re.compile(r"\bUNION\b\s+SELECT", re.IGNORECASE),
    re.compile(r"--\s", re.IGNORECASE),
    re.compile(r"\bTRUNCATE\b\s", re.IGNORECASE),
]


@dataclass(frozen=True, slots=True)
class AuditEntry:
    statement: str
    operation: str  # INSERT/UPDATE/DELETE/OTHER
    param_count: int
    suspicious: bool
    suspicious_patterns: list[str]
    recorded_at: int


class SqlAuditLogger:
    """Persist write-operation audit entries and flag suspicious patterns."""

    def __init__(self, db: Database) -> None:
        self._db = db

    async def record(self, statement: str, parameters) -> AuditEntry:
        operation = self._classify(statement)
        if operation == "OTHER":
            return AuditEntry(statement[:200], operation, 0, False, [], int(time.time()))

        suspicious_hits = [p.pattern for p in SUSPICIOUS_PATTERNS if p.search(statement)]
        entry = AuditEntry(
            statement=statement[:500],
            operation=operation,
            param_count=len(parameters) if hasattr(parameters, "__len__") else 0,
            suspicious=bool(suspicious_hits),
            suspicious_patterns=suspicious_hits,
            recorded_at=int(time.time()),
        )
        await self._db.execute(
            """
            INSERT INTO sql_audit_log
                (statement, operation, param_count, suspicious, suspicious_patterns, recorded_at)
            VALUES (?, ?, ?, ?, ?, ?)
            """,
            (
                entry.statement,
                entry.operation,
                entry.param_count,
                int(entry.suspicious),
                ",".join(entry.suspicious_patterns),
                entry.recorded_at,
            ),
        )
        if entry.suspicious:
            import structlog
            structlog.get_logger(__name__).warning(
                "suspicious_sql_detected",
                statement=entry.statement,
                patterns=entry.suspicious_patterns,
            )
        return entry

    @staticmethod
    def _classify(statement: str) -> str:
        stripped = statement.lstrip().upper()
        for op in ("INSERT", "UPDATE", "DELETE", "CREATE", "ALTER", "DROP"):
            if stripped.startswith(op):
                return op
        return "OTHER"
```

```python
# src/multiscribe_agent/observability/alerts.py
"""Configurable alert rule engine evaluating metrics against thresholds."""

from __future__ import annotations

import asyncio
import time
from collections import deque
from dataclasses import dataclass, field
from pathlib import Path
from typing import Awaitable, Callable, Literal

import yaml

AlertCallback = Callable[[str, dict], Awaitable[None]]
RuleType = Literal["threshold", "window", "ratio"]


@dataclass(frozen=True, slots=True)
class AlertRule:
    name: str
    metric: str
    type: RuleType
    threshold: float
    window_seconds: int = 60
    description: str = ""


@dataclass
class AlertEngine:
    """Evaluate rules against a metric stream and fire callbacks on breach."""

    rules: list[AlertRule]
    _values: dict[str, deque[tuple[float, float]]] = field(default_factory=dict)
    _callbacks: list[AlertCallback] = field(default_factory=list)

    def add_callback(self, callback: AlertCallback) -> None:
        self._callbacks.append(callback)

    def record(self, metric: str, value: float) -> None:
        """Push one metric sample; rules for this metric will be checked."""
        now = time.monotonic()
        if metric not in self._values:
            self._values[metric] = deque(maxlen=10000)
        self._values[metric].append((now, value))
        self._evaluate(metric)

    def _evaluate(self, metric: str) -> None:
        for rule in self.rules:
            if rule.metric != metric:
                continue
            if self._breached(rule):
                asyncio.create_task(self._fire(rule, metric))

    def _breached(self, rule: AlertRule) -> bool:
        samples = list(self._values.get(rule.metric, []))
        if not samples:
            return False
        now = time.monotonic()
        if rule.type == "threshold":
            return samples[-1][1] > rule.threshold
        if rule.type == "window":
            recent = [v for ts, v in samples if now - ts <= rule.window_seconds]
            if not recent:
                return False
            avg = sum(recent) / len(recent)
            return avg > rule.threshold
        if rule.type == "ratio":
            recent = [v for ts, v in samples if now - ts <= rule.window_seconds]
            if len(recent) < 2:
                return False
            breached = sum(1 for v in recent if v > rule.threshold)
            return breached / len(recent) > 0.5
        return False

    async def _fire(self, rule: AlertRule, metric: str) -> None:
        payload = {
            "rule": rule.name,
            "metric": metric,
            "threshold": rule.threshold,
            "description": rule.description,
            "timestamp": int(time.time()),
        }
        for cb in self._callbacks:
            try:
                await cb(rule.name, payload)
            except Exception:
                pass  # callback 失败隔离


def load_rules(path: Path) -> list[AlertRule]:
    raw = yaml.safe_load(path.read_text(encoding="utf-8"))
    return [
        AlertRule(
            name=item["name"],
            metric=item["metric"],
            type=item.get("type", "threshold"),
            threshold=float(item["threshold"]),
            window_seconds=int(item.get("window_seconds", 60)),
            description=item.get("description", ""),
        )
        for item in raw.get("rules", [])
    ]
```

### T4：P1-12 CSRF 保护（`api/middleware/csrf.py` 新增）

**问题定位**：Web 控制台用 cookie 携带 JWT（hash router），存在 CSRF 风险。

**修复**：采用 **double-submit cookie** 模式（无需服务端 session 存储）。

```python
# src/multiscribe_agent/api/middleware/csrf.py
"""Double-submit cookie CSRF protection for browser-submitted state changes."""

from __future__ import annotations

import secrets

from starlette.middleware.base import BaseHTTPMiddleware
from starlette.requests import Request
from starlette.responses import JSONResponse, Response


CSRF_COOKIE_NAME = "multiscribe_csrf"
CSRF_HEADER_NAME = "X-CSRF-Token"
SAFE_METHODS = frozenset({"GET", "HEAD", "OPTIONS", "TRACE"})
STATE_CHANGING_METHODS = frozenset({"POST", "PUT", "PATCH", "DELETE"})


class CsrfMiddleware(BaseHTTPMiddleware):
    """Validate X-CSRF-Token header against cookie for state-changing requests."""

    def __init__(self, app, *, exempt_paths: frozenset[str] = frozenset()) -> None:
        super().__init__(app)
        self._exempt = exempt_paths

    async def dispatch(self, request: Request, call_next):
        # 总是为新会话种 cookie
        response = await self._handle(request, call_next)
        if not request.cookies.get(CSRF_COOKIE_NAME):
            token = secrets.token_urlsafe(32)
            response.set_cookie(
                CSRF_COOKIE_NAME,
                token,
                httponly=False,  # 前端 JS 需读取
                samesite="lax",
                secure=request.url.scheme == "https",
            )
        return response

    async def _handle(self, request: Request, call_next) -> Response:
        if request.method in SAFE_METHODS:
            return await call_next(request)
        if request.url.path in self._exempt:
            return await call_next(request)

        # Bearer token 认证（API 客户端）豁免 CSRF
        auth_header = request.headers.get("Authorization", "")
        if auth_header.startswith("Bearer "):
            return await call_next(request)

        if request.method in STATE_CHANGING_METHODS:
            cookie_token = request.cookies.get(CSRF_COOKIE_NAME)
            header_token = request.headers.get(CSRF_HEADER_NAME)
            if not cookie_token or not header_token or cookie_token != header_token:
                return JSONResponse(
                    status_code=403,
                    content={"detail": "CSRF token missing or mismatched"},
                )
        return await call_next(request)
```

```typescript
// frontend/src/services/api.ts  追加 CSRF token 读取：

function getCsrfToken(): string | null {
  const match = document.cookie.match(/multiscribe_csrf=([^;]+)/)
  return match ? match[1] : null
}

// 在 request<T> 中，对 POST/PUT/PATCH/DELETE 自动注入：
if (options.method && !["GET", "HEAD"].includes(options.method)) {
  const csrf = getCsrfToken()
  if (csrf) headers["X-CSRF-Token"] = csrf
}
```
