### T1：P0-1 上下文溢出保护（`agents/context.py`）

**问题定位**：`add_user()` 只 append + trim，但 `trim_if_needed` 裁的是**历史消息**，当前 user 消息本身若超过 `token_budget`，trim 后仍超窗。

**修复**：

```python
# src/multiscribe_agent/agents/context.py  追加常量与截断逻辑

USER_MESSAGE_BUDGET_RATIO = 0.8  # 单条 user 消息最多占 80% 预算


def add_user(self, message: str) -> None:
    """Append a user message, truncating oversized input before trim."""
    truncated = self._maybe_truncate_user_message(message)
    self.messages.append(AIMessage(role="user", content=truncated))
    self.trim_if_needed()


def _maybe_truncate_user_message(self, message: str) -> str:
    """Truncate user input that alone would exceed 80% of the token budget."""
    estimated = self._estimate_text(message)
    threshold = int(self.token_budget * USER_MESSAGE_BUDGET_RATIO)
    if estimated <= threshold:
        return message
    # 保留头部 + 尾部，中间用占位符
    char_limit = threshold * 4  # 反向估算字符数（4 char ≈ 1 token）
    head = message[: char_limit // 2]
    tail = message[-char_limit // 2 :]
    return f"{head}\n\n[... content truncated: {estimated} tokens > {threshold} budget ...]\n\n{tail}"


def should_warn_budget(self, threshold: float = 0.8) -> bool:
    """Return True when cumulative token usage crosses the budget ratio."""
    if self.token_budget <= 0:
        return False
    current = self.estimate_tokens(self.build_messages(trim=False))
    return current >= int(self.token_budget * threshold)


def estimated_tokens_remaining(self) -> int:
    """Return the approximate remaining tokens before the budget is hit."""
    current = self.estimate_tokens(self.build_messages(trim=False))
    return max(0, self.token_budget - current)


# 拆出 _estimate_text 以便单测直接调用
def _estimate_text(self, text: str) -> int:
    """Estimate token count for a raw string (4 chars ≈ 1 token heuristic)."""
    return (len(text) + MESSAGE_OVERHEAD_TOKENS + 3) // 4
```

**注意**：`_estimate_text` 从 `_estimate_message` 中抽出公共部分，避免重复实现。

### T2：P0-2 工作流超时保护（`agents/workflow/engine.py`）

**问题定位**：`stream()` 无超时；嵌套子工作流或 Loop 死循环会无限挂起。

**修复**：

```python
# src/multiscribe_agent/agents/workflow/engine.py  修改 stream() 签名与实现

DEFAULT_WORKFLOW_TIMEOUT_SECONDS = 300.0


async def stream(
    self,
    workflow_id: str,
    input_data: object,
    *,
    date: str | None = None,
    timeout: float = DEFAULT_WORKFLOW_TIMEOUT_SECONDS,
) -> AsyncIterator[WorkflowEvent]:
    """Yield workflow events; abort with workflow_error if timeout elapses."""
    del date
    raw = await self._workflow_store.get("workflows", workflow_id)
    if raw is None:
        raise WorkflowError(f"workflow not found: {workflow_id}")
    trace_id = uuid4().hex
    results: dict[str, object] = {"start": input_data}
    yield WorkflowEvent("workflow_start", {"workflow_id": workflow_id, "timeout": timeout}, trace_id)
    try:
        graph = build_graph(WorkflowDefinition.model_validate(raw))
        levels = topological_levels(graph)
    except WorkflowError as exc:
        yield WorkflowEvent("workflow_error", {"message": str(exc)}, trace_id)
        return

    deadline = asyncio.get_event_loop().time() + timeout
    try:
        for level in levels:
            # 检查是否已超时
            remaining = deadline - asyncio.get_event_loop().time()
            if remaining <= 0:
                raise asyncio.TimeoutError()
            executions: list[tuple[WorkflowStep, object]] = []
            for step_id in level:
                step = graph.steps[step_id]
                step_input = self._input(step, graph.predecessors[step_id], results)
                executions.append((step, step_input))
                yield WorkflowEvent("step_start", {"step_id": step_id}, trace_id)
            outcomes = await asyncio.wait_for(
                asyncio.gather(
                    *(
                        self._execute_step(step, step_input, trace_id)
                        for step, step_input in executions
                    ),
                    return_exceptions=True,
                ),
                timeout=remaining,
            )
            # ... 既有 outcomes 处理逻辑 ...
    except asyncio.TimeoutError:
        yield WorkflowEvent(
            "workflow_error",
            {"message": f"workflow timed out after {timeout}s", "timeout": timeout},
            trace_id,
        )
        return
```

**设计要点**：
- 用 `asyncio.wait_for` 包裹每层 `gather`，剩余时间作为该层预算。
- 超时不抛裸 `TimeoutError`，而是 yield `workflow_error` 事件（与既有错误流一致）。
- `run()` 调用方仍会从 `workflow_error` 抛 `WorkflowError`，行为兼容。

### T3：P0-3 分布式 Trace 传播（`observability/trace_headers.py` 新增）

**问题定位**：`tracer.py` 已启动 OTel Provider，但出站 HTTP（`httpx` / `aiohttp`）不注入 trace context，跨服务调用链断裂。

**修复**：

```python
# src/multiscribe_agent/observability/trace_headers.py
"""Inject OTel trace context into outbound HTTP headers."""

from __future__ import annotations

from importlib import import_module
from typing import Any, MutableMapping


TRACE_HEADER = "X-Trace-Id"


def inject_trace_headers(headers: MutableMapping[str, str]) -> MutableMapping[str, str]:
    """Add X-Trace-Id when an OTel span is active; return headers unchanged otherwise."""
    try:
        trace = import_module("opentelemetry.trace")
    except ImportError:
        return headers
    span = trace.get_current_span()
    context = span.get_span_context()
    if not context.is_valid:
        return headers
    headers[TRACE_HEADER] = format(context.trace_id, "032x")
    return headers


def extract_trace_id_from_headers(headers: Any) -> str | None:
    """Read an inbound X-Trace-Id header (case-insensitive) for log correlation."""
    if headers is None:
        return None
    getter = headers.get if hasattr(headers, "get") else None
    if getter is None:
        return None
    for key in (TRACE_HEADER, TRACE_HEADER.lower(), "X-Trace-Id".lower()):
        value = getter(key)
        if isinstance(value, str) and value.strip():
            return value.strip()
    return None
```

**接线点**（`llm/provider.py` 的 HTTP 客户端构造处）：

```python
# src/multiscribe_agent/llm/provider.py  在构造 httpx.AsyncClient / ChatOpenAI 时：
```

```python
    inject_trace_headers(request.headers)

    event_hooks={"request": [_trace_injection_hook]},
    ...
)
```

### T4：P0-4 速率限制 middleware（`api/middleware/rate_limit.py` 新增）

**问题定位**：`/api/ai/v1/*` 已有 interop 限流（P22），但人类用户端点（`/api/agents/run`、`/api/digest/run`、`/api/auth/login`）无限流。

**设计**：复用既有 `SlidingWindowLimiter`（P22 已实现），包一层 Starlette middleware。

```python
# src/multiscribe_agent/api/middleware/rate_limit.py
"""Per-path rate limiting middleware for human-facing API endpoints."""

from __future__ import annotations

from collections import defaultdict, deque
from time import monotonic
from typing import Iterable

from starlette.middleware.base import BaseHTTPMiddleware
from starlette.requests import Request
from starlette.responses import JSONResponse


class EndpointRateLimiter(BaseHTTPMiddleware):
    """Sliding-window per-IP rate limiter keyed by configured path prefixes."""

    def __init__(
        self,
        app,
        *,
        rules: dict[str, tuple[int, int]],  # path_prefix → (limit, window_seconds)
        exempt_paths: Iterable[str] = (),
    ) -> None:
        super().__init__(app)
        self._rules = sorted(rules.items(), key=lambda kv: len(kv[0]), reverse=True)
        self._exempt = frozenset(exempt_paths)
        self._hits: dict[str, deque[float]] = defaultdict(deque)

    async def dispatch(self, request: Request, call_next):
        path = request.url.path
        if path in self._exempt:
            return await call_next(request)

        limit, window = self._match_rule(path)
        if limit is None:
            return await call_next(request)

        client_ip = request.client.host if request.client else "unknown"
        key = f"{client_ip}:{path}"
        now = monotonic()
        cutoff = now - window
        hits = self._hits[key]
        while hits and hits[0] < cutoff:
            hits.popleft()
        if len(hits) >= limit:
            retry_after = int(window - (now - hits[0])) + 1
            return JSONResponse(
                status_code=429,
                content={"detail": f"rate limit exceeded: {limit}/{window}s", "retry_after": retry_after},
                headers={"Retry-After": str(retry_after)},
            )
        hits.append(now)
        return await call_next(request)

    def _match_rule(self, path: str) -> tuple[int | None, int]:
        for prefix, (limit, window) in self._rules:
            if path.startswith(prefix):
                return limit, window
        return None, 0
```
