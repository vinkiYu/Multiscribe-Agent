### T1：P1-7 SQLite 连接池（`infra/connection_pool.py` 新增）

**问题定位**：`Database` 单连接，高并发读场景瓶颈。

**设计**：SQLite WAL 模式下允许多读并发 + 单写串行。连接池维护 N 个读连接，写操作走专用写连接。

```python
# src/multiscribe_agent/infra/connection_pool.py
"""SQLite WAL-aware connection pool with separate read and write lanes."""

from __future__ import annotations

import asyncio
from contextlib import asynccontextmanager
from dataclasses import dataclass, field

import aiosqlite


@dataclass
class PoolStats:
    acquires: int = 0
    releases: int = 0
    read_waits: int = 0
    write_waits: int = 0


class ConnectionPool:
    """Bounded pool of read connections + a single exclusive write connection."""

    def __init__(
        self,
        db_path: str,
        *,
        read_pool_size: int = 5,
        write_timeout: float = 30.0,
    ) -> None:
        self._db_path = db_path
        self._read_pool_size = read_pool_size
        self._write_timeout = write_timeout
        self._read_sem = asyncio.Semaphore(read_pool_size)
        self._write_lock = asyncio.Lock()
        self._read_connections: list[aiosqlite.Connection] = []
        self._write_connection: aiosqlite.Connection | None = None
        self._initialized = False
        self.stats = PoolStats()

    async def initialize(self) -> None:
        if self._initialized:
            return
        for _ in range(self._read_pool_size):
            conn = await self._open_connection()
            await conn.execute("PRAGMA query_only = ON")  # 读连接只读
            self._read_connections.append(conn)
        self._write_connection = await self._open_connection()
        await self._write_connection.execute("PRAGMA query_only = OFF")
        self._initialized = True

    async def _open_connection(self) -> aiosqlite.Connection:
        conn = await aiosqlite.connect(self._db_path)
        await conn.execute("PRAGMA journal_mode = WAL")
        await conn.execute("PRAGMA busy_timeout = 5000")
        await conn.row_factory.__class__  # 触发属性检查
        conn.row_factory = aiosqlite.Row
        return conn

    @asynccontextmanager
    async def acquire_read(self):
        """Acquire one read connection; blocks if pool is exhausted."""
        async with self._read_sem:
            self.stats.acquires += 1
            conn = self._read_connections.pop()
            try:
                yield conn
            finally:
                self._read_connections.append(conn)
                self.stats.releases += 1

    @asynccontextmanager
    async def acquire_write(self):
        """Acquire the exclusive write connection; serializes all writes."""
        async with self._write_lock:
            self.stats.acquires += 1
            if self._write_connection is None:
                raise RuntimeError("write connection not initialized")
            try:
                yield self._write_connection
            finally:
                self.stats.releases += 1

    async def close(self) -> None:
        for conn in self._read_connections:
            await conn.close()
        self._read_connections.clear()
        if self._write_connection is not None:
            await self._write_connection.close()
            self._write_connection = None
        self._initialized = False
```

**Database 改造**：保持 `Database` 类既有方法签名（`execute/fetchall/fetchone`），内部路由到 pool 的 write/read lane。

```python
# src/multiscribe_agent/infra/db.py  Database 类改造：

class Database:
    def __init__(self, pool: ConnectionPool | None = None, conn=None):
        self._pool = pool
        self._single_conn = conn  # 兼容旧的单连接模式（测试用）

    async def execute(self, statement, parameters=()):
        if self._pool is not None:
            async with self._pool.acquire_write() as conn:
                return await conn.execute(statement, parameters)
        return await self._single_conn.execute(statement, parameters)

    async def fetchall(self, statement, parameters=()):
        if self._pool is not None:
            async with self._pool.acquire_read() as conn:
                return await conn.fetchall(statement, parameters)
        return await self._single_conn.fetchall(statement, parameters)
```

### T2：P1-8 FTS5 中文分词（`infra/text_tokenize.py` 新增）

**问题定位**：FTS5 默认 `unicode61` 对中文按字切分，召回率低（搜"大模型"匹配不到"大语言模型"）。

**修复**：写入 FTS5 前用 jieba 分词，词间加空格。

```python
# src/multiscribe_agent/infra/text_tokenize.py
"""Chinese-aware tokenizer with graceful jieba degradation."""

from __future__ import annotations

from functools import lru_cache


@lru_cache(maxsize=1)
def _get_jieba():
    try:
        import jieba  # type: ignore[import-not-found]
        # 静默加载词典
        jieba.setLogLevel(20)  # 抑制 jieba 初始化日志
        return jieba
    except ImportError:
        return None


def tokenize_for_fts(text: str) -> str:
    """Tokenize text for FTS5 indexing; uses jieba for CJK, falls back to original."""
    if not text:
        return ""
    jieba = _get_jieba()
    if jieba is None:
        return text  # 降级：原样返回，FTS5 用 unicode61
    tokens = jieba.cut_for_search(text)
    return " ".join(token.strip() for token in tokens if token.strip())


def is_chinese_tokenization_available() -> bool:
    """Return True if jieba is importable."""
    return _get_jieba() is not None
```

### T4：P1-10 插件沙箱隔离（`plugins/sandbox.py` 新增，可选启用）

**问题定位**：内置插件直接 import；第三方插件可能污染全局状态。

**设计**：沙箱通过 subprocess 执行，stdin/stdout JSON 协议。**默认不启用**（性能开销），仅对 `metadata.sandboxed = True` 的插件启用。

```python
# src/multiscribe_agent/plugins/sandbox.py
"""Subprocess-isolated plugin executor for untrusted third-party plugins."""

from __future__ import annotations

import asyncio
import json
import sys
from dataclasses import dataclass
from pathlib import Path
from typing import Any


@dataclass(frozen=True)
class SandboxConfig:
    plugin_path: Path
    timeout_seconds: float = 30.0
    memory_limit_mb: int = 256


class SandboxError(RuntimeError):
    """Raised when a sandboxed plugin fails, times out, or returns invalid output."""


class SandboxedPluginExecutor:
    """Run a plugin in a child process with stdin/stdout JSON protocol."""

    def __init__(self, config: SandboxConfig) -> None:
        self._config = config

    async def execute(self, method: str, arguments: dict[str, Any]) -> dict[str, Any]:
        """Invoke one plugin method via subprocess and return its JSON response."""
        payload = json.dumps({"method": method, "arguments": arguments})
        try:
            proc = await asyncio.create_subprocess_exec(
                sys.executable,
                str(self._config.plugin_path),
                stdin=asyncio.subprocess.PIPE,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE,
            )
        except OSError as exc:
            raise SandboxError(f"failed to spawn plugin: {exc}") from exc

        try:
            stdout, stderr = await asyncio.wait_for(
                proc.communicate(payload.encode("utf-8")),
                timeout=self._config.timeout_seconds,
            )
        except asyncio.TimeoutError:
            proc.kill()
            raise SandboxError(
                f"plugin timed out after {self._config.timeout_seconds}s"
            ) from None

        if proc.returncode != 0:
            raise SandboxError(
                f"plugin exited {proc.returncode}: {stderr.decode('utf-8', errors='replace')[:500]}"
            )
        try:
            result = json.loads(stdout.decode("utf-8"))
        except json.JSONDecodeError as exc:
            raise SandboxError(f"plugin returned invalid JSON: {exc}") from exc
        if not isinstance(result, dict):
            raise SandboxError("plugin must return a JSON object")
        return result
```

**关键约束**：
- 沙箱是**可选**的；内置插件仍走进程内调用。
- `PluginMetadata` 可新增 `sandboxed: bool = False` 字段（本包不改 metadata schema，留 follow-up）。
